## Name Capitalizer
```
def format_name(first_name, last_name):
  # return f"""{first_name} {last_name}""".title()
  first_name = first_name[0].capitalize() + first_name[1:].lower()
  last_name = last_name[0].capitalize() + last_name[1:].lower()
  return first_name + ' ' + last_name


full_name = format_name('elvis', 'presley')
print(full_name)
```

## Days in Month
```
def is_leap(year):
  if year % 4 == 0:
    if year % 100 == 0:
      if year % 400 == 0:
        return True
      else:
        return False
    else:
      return True
  else:
    return False

def days_in_month(year, month):
  month_days = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]  
  if is_leap(year) and int(month) == 2:
    return 29
  return month_days[int(month - 1)]
  
```
